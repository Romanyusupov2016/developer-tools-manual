java:
sudo apt-get update - получить новые списки пакетов.
sudo apt-get upgrade - выполнить обновление установленных пакетов.
sudo apt install default-jre - установка JRE.
sudo apt install default-jdk - установка JDK.
java -version
javac -version - узнать версию установленной java.

intellij-idea:
tar -zxvf ideaIC-version.tar.gz - распаковка архива.
idea-IC-version/bin/idea.sh - запуск скрипта установки intellij-idea.
