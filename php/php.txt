sudo apt-get update
sudo apt install apache2 - установка Apache.
sudo apt install mysql-server - установка MySQL.
sudo apt install php libapache2-mod-php php-mysql - установка PHP.

php -S localhost:8000 - запуск встроенного сервера.

/var/www/html/ - рабочий каталог Apache.
/etc/php/7.2/apache2/php.ini - файл конфигурации PHP.

sudo systemctl start apache2.service - запуск Apache.
sudo systemctl stop apache2.service - остановка Apache.
sudo systemctl restart apache2.service - перезапуск Apache.
